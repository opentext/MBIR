    actuate.initialize(bireportingjsURL, reqOps, "", "", initViewer);
}

function initViewer(option) {
	var newconfig = new actuate.dataprovision.viewer.UIConfig();
	newconfig.setHideHeader(false);
	var dpViewer = new actuate.dataprovision.Viewer("dpContainer",newconfig);
	dpViewer.setWidth(1024);
	dpViewer.setHeight(768);
	dpViewer.submit();
}
